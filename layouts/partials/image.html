{{- /* Responsive for a raster image resource. Params: image (resource, required) alt (string) sizes (string, optional - defaults to the content column) loading (string, optional - "lazy" by default, use "eager" above the fold) class (string, optional - added to the ) Generates variants at several widths, never upscaling, with intrinsic width/height to avoid layout shift. On the Hugo extended edition the variants are WebP (smaller, faster); on the standard edition they stay in the image's original format and a one-time warning is logged, so the build still succeeds. */ -}} {{- $img := .image -}} {{- $alt := .alt | default "" -}} {{- $sizes := .sizes | default "(min-width: 48rem) 768px, calc(100vw - 2.5rem)" -}} {{- $loading := .loading | default "lazy" -}} {{- $class := .class | default "" -}} {{- /* WebP encoding needs the extended edition. The format token drops straight into Resize: "…x webp q82" on extended, "…x q82" (original format) else. */ -}} {{- $webpOK := hugo.IsExtended -}} {{- $fmt := cond $webpOK "webp " "" -}} {{- /* Cap the largest delivered width. */ -}} {{- $max := 1600 -}} {{- $src := $img -}} {{- if gt $img.Width $max -}} {{- $src = $img.Resize (printf "%dx q82" $max) -}} {{- end -}} {{- /* Widths to generate: standard steps below the source, plus the source. */ -}} {{- $widths := slice -}} {{- range (slice 480 768 1024 1600) -}} {{- if lt . $src.Width -}}{{- $widths = $widths | append . -}}{{- end -}} {{- end -}} {{- $widths = $widths | append $src.Width -}} {{- $set := slice -}} {{- range $w := $widths -}} {{- $set = $set | append (printf "%s %dw" ($src.Resize (printf "%dx %sq82" $w $fmt)).RelPermalink $w) -}} {{- end -}} {{- /* A modest fallback src for browsers without srcset support. */ -}} {{- $fw := cond (gt $src.Width 768) 768 $src.Width -}} {{- $fallback := $src.Resize (printf "%dx %sq82" $fw $fmt) -}} {{- /* Warn once per build if we couldn't produce WebP. */ -}} {{- if and (not $webpOK) (not (hugo.Store.Get "warnedNoWebP")) -}} {{- hugo.Store.Set "warnedNoWebP" true -}} {{- warnf "This Hugo isn't the extended edition, so images are served in their original format instead of smaller WebP. Install Hugo extended for faster-loading images: https://gohugo.io/installation/" -}} {{- end -}}
