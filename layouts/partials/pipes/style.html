{{- /* Build the site stylesheet by concatenating these files in cascade order. Small changes go in the sheet they belong to; a substantial, self-contained component earns its own css/components/.css, added to this list like an @import. Development: served as-is for readable devtools. */ -}} {{- $sheets := slice (resources.Get "css/vendor/modern-normalize.css") (resources.Get "css/variables.css") (resources.Get "css/components/fonts.css") (resources.Get "css/reset.css") (resources.Get "css/components/scaffolding.css") (resources.Get "css/vendor/syntax.css") (resources.Get "css/components/layout.css") (resources.Get "css/components/navigation.css") -}} {{- $css := $sheets | resources.Concat "css/bundle.css" -}} {{- if hugo.IsProduction -}} {{- $css = $css | resources.Minify | resources.Fingerprint "sha256" -}} {{- end -}}
