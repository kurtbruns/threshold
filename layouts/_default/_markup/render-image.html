{{- /* Markdown image render hook: ![alt](photo.jpg) becomes a responsive, optimized when the image is a local raster file. External URLs, SVGs, and animated GIFs pass through as a plain . */ -}} {{- $u := .Destination -}} {{- $alt := .PlainText -}} {{- $img := or (.PageInner.Resources.GetMatch $u) (resources.Get $u) -}} {{- if and $img (in (slice "jpeg" "png" "webp" "tiff" "bmp") $img.MediaType.SubType) -}} {{ partial "image.html" (dict "image" $img "alt" $alt) }} {{- else -}} {{- end -}}
