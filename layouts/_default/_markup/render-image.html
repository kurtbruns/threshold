{{- /* Markdown image render hook: ![alt](photo.jpg). src resolves the same way as the image shortcode: the page's bundle first, then assets/. A local raster file (jpeg/png/webp/tiff/bmp) runs through the image.html partial for responsive output; anything else (external URLs, SVG, animated GIF) passes through as a plain . */ -}} {{- $u := .Destination -}} {{- $alt := .PlainText -}} {{- $img := or (.PageInner.Resources.GetMatch $u) (resources.Get $u) -}} {{- if and $img (in (slice "jpeg" "png" "webp" "tiff" "bmp") $img.MediaType.SubType) -}} {{ partial "image.html" (dict "image" $img "alt" $alt) }} {{- else -}} {{- end -}}
